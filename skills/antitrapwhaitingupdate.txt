## 1. Process Management & Timeouts

### ✅ Safe Timeout Wrappers

```bash
# Basic timeout (preferred method)
timeout 60s <command>

# Timeout with fallback kill
timeout --kill-after=5s 60s <command>

# Background process with safe cleanup
run_with_timeout() {
    local timeout=$1
    shift
    "$@" & 
    local pid=$!
    ( sleep $timeout; kill -0 $pid 2>/dev/null && kill -9 $pid ) &
    local killer=$!
    wait $pid 2>/dev/null
    local status=$?
    kill -9 $killer 2>/dev/null
    return $status
}

# Usage
run_with_timeout 60 nmap -p- target.com
```

### ❌ Avoid These Patterns

```bash
# WRONG: Race condition
<command> & PID=$!; sleep 60; kill -9 $PID 2>/dev/null

# WRONG: Orphaned processes
<command> & sleep 60; killall <command>
```

## 9. Adaptive Network Timeouts

### Dynamic Timeout Calculation

```bash
# Measure RTT and set adaptive timeout
calculate_timeout() {
    local target=$1
    
    # Ping 5 times and get average RTT
    local rtt=$(ping -c 5 -W 2 $target 2>/dev/null | \
                grep "rtt" | \
                awk -F'/' '{print $5}')
    
    if [ -z "$rtt" ]; then
        echo "15"  # Default fallback
        return
    fi
    
    # Timeout = max(10s, 3 * RTT)
    local timeout=$(echo "$rtt * 3 / 1000" | bc)
    if (( $(echo "$timeout < 10" | bc -l) )); then
        echo "10"
    else
        echo "$timeout"
    fi
}

# Usage
TIMEOUT=$(calculate_timeout target.com)
curl --max-time $TIMEOUT http://target.com/api
```

import subprocess
import requests
import statistics

def measure_latency(host, samples=5):
    """Measure network latency to target"""
    try:
        result = subprocess.run(
            ['ping', '-c', str(samples), '-W', '2', host],
            capture_output=True, text=True, timeout=15
        )
        
        # Parse RTT values
        rtts = []
        for line in result.stdout.split('\n'):
            if 'time=' in line:
                rtt = float(line.split('time=')[1].split()[0])
                rtts.append(rtt)
        
        if rtts:
            return statistics.mean(rtts) / 1000  # Convert to seconds
    except:
        pass
    
    return 0.5  # Default 500ms

def adaptive_request(url, method='GET', **kwargs):
    """Make HTTP request with adaptive timeout"""
    host = url.split('/')[2]
    latency = measure_latency(host)
    
    # Timeout = max(10s, 3 * latency + 5s buffer)
    timeout = max(10, latency * 3 + 5)
    
    print(f"📊 Measured latency: {latency*1000:.0f}ms → Using timeout: {timeout:.1f}s")
    
    try:
        return requests.request(method, url, timeout=(5, timeout), **kwargs)
    except requests.Timeout:
        print(f"🚨 Timeout after {timeout}s - possible tarpit")
        return None

# Usage
